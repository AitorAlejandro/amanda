isPrimitive


            import { isPrimitive } from '@aitoralejandro/amanda';
            // returns true is argument is a primitive js type

            // examples
            isPrimitive('a string'); // returns true
            isPrimitive(false); // returns true
            isPrimitive([1, 2, 3]); // returns false
            isPrimitive({}); // returns true

isArray


            import { isArray } from '@aitoralejandro/amanda';
            // returns true is the argument is an array

            // examples
            isArray('a string'); // returns false
            isArray(false); // returns false
            isArray([1, 2, 3]); // returns true
            isArray({}); // returns false

isEmptyArray


            import { isEmptyArray } from '@aitoralejandro/amanda';
            // returns true is the array have no elements

            // examples
            isEmptyArray([1, 2, 3]); // returns false
            isEmptyArray([]); // returns true
            isEmptyArray({}); // throws an error because is not an array
