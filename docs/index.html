@aitoralejandro/amanda

amanda is a library of utilities that will help you focus on the important things in your project and prevent you from wasting time on minor things

capitalize


            import { capitalize } from '@aitoralejandro/amanda';
            // returns the passed string with the first letter in upper case

            // examples
            capitalize('sleepover'); // 'Sleepover'
            capitalize('hi Amanda'); // 'Hi Amanda'

decapitalize


            import { decapitalize } from '@aitoralejandro/amanda';
            // returns the passed string with the first letter in lower case

            // examples
            decapitalize('Sleepover'); // 'sleepover'
            decapitalize('Hi Amanda'); // 'hi Amanda'

isArray


            import { isArray } from '@aitoralejandro/amanda';
            // returns true is the argument is an array

            // examples
            isArray('a string'); // returns false
            isArray(false); // returns false
            isArray([1, 2, 3]); // returns true
            isArray({}); // returns false

isBoolean


            import { isBoolean } from '@aitoralejandro/amanda';
            // returns true is the argument is boolean type

            // examples
            isBoolean(true); // returns true
            isBoolean(false); // returns true
            isBoolean([1, 2, 3]); // returns false

isEmptyArray


            import { isEmptyArray } from '@aitoralejandro/amanda';
            // returns true is the array have no elements

            // examples
            isEmptyArray([1, 2, 3]); // returns false
            isEmptyArray([]); // returns true
            isEmptyArray({}); // throws an error because is not an array

isNumber


            import { isNumber } from '@aitoralejandro/amanda';
            // returns true is the argument is a number

            // examples
            isNumber(0); // returns true
            isNumber(123); // returns true
            isNumber(123.456); // returns true
            isNumber(NaN); // returns false
            isNumber(Infinite); // returns false
            isNumber('a string'); // returns false
